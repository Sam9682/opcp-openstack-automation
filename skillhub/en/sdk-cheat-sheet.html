SDK & API Cheat Sheet

Difficulty: Intermediate · Estimated time: 10 min

A quick-reference page for every principal OpenStack SDK call used across the labs. Each section maps to one of the project's managers and shows the underlying openstacksdk method alongside a short usage example.

1. Authentication

Handled by openstack_sdk/auth_manager.py.

ActionSDK Call
Connectopenstack.connect(**params)
Authorize (validate token)conn.authorize()
Close connectionconn.close()
 
import openstack

conn = openstack.connect(
    auth_url="https://auth.cloud.ovh.net/v3",
    project_name="my-project",
    username="user",
    password="secret",
    region_name="GRA7",
    user_domain_name="Default",
    project_domain_name="Default",
)
conn.authorize()   # validates the token
# … use conn …
conn.close()

Application Credentials

conn = openstack.connect(
    auth_url="https://auth.cloud.ovh.net/v3",
    region_name="GRA7",
    application_credential_id="APP_CRED_ID",
    application_credential_secret="APP_CRED_SECRET",
)

2. Networking

Handled by openstack_sdk/network_manager.py.

ActionSDK Call
Create networkconn.network.create_network()
Create subnetconn.network.create_subnet()
Get networkconn.network.get_network(id)
List networksconn.network.networks()
List subnetsconn.network.subnets()
Delete networkconn.network.delete_network(id)
Delete subnetconn.network.delete_subnet(id)
 
# Create a private network + subnet
network = conn.network.create_network(
    name="my-network",
    admin_state_up=True,
)

subnet = conn.network.create_subnet(
    name="my-subnet",
    network_id=network.id,
    cidr="192.168.1.0/24",
    ip_version=4,
    enable_dhcp=True,
    dns_nameservers=["213.186.33.99"],
    gateway_ip="192.168.1.1",
)

# List all networks
for net in conn.network.networks():
    print(net.name, net.id, net.status)

3. Security Groups

Handled by openstack_sdk/security_group_manager.py.

ActionSDK Call
Create security groupconn.network.create_security_group()
Create ruleconn.network.create_security_group_rule()
List security groupsconn.network.security_groups()
Delete security groupconn.network.delete_security_group(id)
 
# Create a security group with SSH + HTTP rules
sg = conn.network.create_security_group(
    name="web-sg",
    description="Allow SSH and HTTP",
)

# Allow SSH (port 22)
conn.network.create_security_group_rule(
    security_group_id=sg.id,
    direction="ingress",
    ethertype="IPv4",
    protocol="tcp",
    port_range_min=22,
    port_range_max=22,
    remote_ip_prefix="0.0.0.0/0",
)

# Allow HTTP (port 80)
conn.network.create_security_group_rule(
    security_group_id=sg.id,
    direction="ingress",
    ethertype="IPv4",
    protocol="tcp",
    port_range_min=80,
    port_range_max=80,
    remote_ip_prefix="0.0.0.0/0",
)

4. Compute

Handled by openstack_sdk/compute_manager.py.

ActionSDK Call
Find flavorconn.compute.find_flavor(name)
Find imageconn.compute.find_image(name)
Create serverconn.compute.create_server()
Get serverconn.compute.get_server(id)

# Resolve flavor and image by name
flavor = conn.compute.find_flavor("s1-2")
image  = conn.compute.find_image("Ubuntu 22.04")

# Launch an instance
server = conn.compute.create_server(
    name="my-instance",
    flavor_id=flavor.id,
    image_id=image.id,
    key_name="my-keypair",
    networks=[{"uuid": network.id}],
    security_groups=[{"name": "web-sg"}],
    metadata={"student_id": "slepetre"},
    user_data="#!/bin/bash\napt update",
)

# Poll until ACTIVE
import time
while True:
    srv = conn.compute.get_server(server.id)
    if srv.status == "ACTIVE":
        break
    time.sleep(5)

5. Volumes (Block Storage)

Handled by openstack_sdk/volume_manager.py.

ActionSDK Call
Create volumeconn.block_storage.create_volume()
Get volumeconn.block_storage.get_volume(id)
List volumesconn.block_storage.volumes()
Delete volumeconn.block_storage.delete_volume(id)
Attach volumeconn.compute.create_volume_attachment()
Detach volumeconn.compute.delete_volume_attachment()
 
# Create a 10 GB volume
volume = conn.block_storage.create_volume(
    name="data-vol",
    size=10,
    volume_type="classic",
)

# Wait until available
while conn.block_storage.get_volume(volume.id).status != "available":
    time.sleep(2)

# Attach to an instance
conn.compute.create_volume_attachment(
    server=server,
    volume_id=volume.id,
)

# Detach later
conn.compute.delete_volume_attachment(volume.id, server.id)

6. Deployment Engine

Handled by openstack_sdk/deployment_engine.py. Orchestrates all managers in dependency order.

ActionMethod
Full deployengine.deploy_infrastructure(config)
Rollbackengine.rollback_resources(resources, conn)
Manual cleanupengine.cleanup_resources(resource_ids)
 
from openstack_sdk.deployment_engine import DeploymentEngine
from config.config_manager import ConfigManager

config_mgr = ConfigManager()
config = config_mgr.load_config("config.yaml")

engine = DeploymentEngine(credentials)
result = engine.deploy_infrastructure(config)

if result.success:
    print(f"Deployed in {result.duration_seconds:.1f}s")
    print(f"Resources: {result.created_resources}")
else:
    print(f"Failed: {result.failed_resources}")
    print(f"Orphaned: {result.orphaned_resources}")

7. Cleanup — Delete Everything

Resources must be deleted in reverse dependency order: volumes → instances → security groups → subnets → networks.

# Quick teardown of known resource IDs
engine.cleanup_resources({
    "volumes":         ["vol-id-1"],
    "instances":       ["srv-id-1"],
    "security_groups": ["sg-id-1"],
    "subnets":         ["sub-id-1"],
    "networks":        ["net-id-1"],
})

Summary

  • openstack.connect() — entry point for every operation.
  • conn.network.* — networks, subnets, security groups & rules.
  • conn.compute.* — flavors, images, servers, volume attachments.
  • conn.block_storage.* — volume lifecycle.
  • DeploymentEngine — orchestrates all of the above with rollback.
